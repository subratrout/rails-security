def decrypt_ssn
  aes = OpenSSL::Cipher::Cipher.new(cipher_type)
  aes.decrypt
  aes.key = key
  aes.iv = iv if iv != nil
  aes.update(self.encrypted_ssn) + aes.final
end

def encrypt_ssn
  aes = OpenSSL::Cipher::Cipher.new(cipher_type)
  aes.encrypt
  aes.key = key
  aes.iv = iv if iv != nil
  self.encrypted_ssn = aes.update(self.SSN) + aes.final
  self.SSN = nil
end
